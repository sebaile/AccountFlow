  <script>
    let cuentas = [];
    let clientes = [];
    let editCuentaId = null;
    let editClienteId = null;

    const $ = (id) => document.getElementById(id);
    const uid = () => 'id_' + Date.now() + '_' + Math.random().toString(36).slice(2, 8);

    // Lista de precios (igual a la hoja "Precios" del Excel real)
    const PRECIOS = {
      'Netflix': { costo: 4000, valor: 5500, ciclo: 'Mensual' },
      'Disney+': { costo: 3000, valor: 4500, ciclo: 'Anual' },
      'HBO Max': { costo: 3000, valor: 4500, ciclo: 'Anual' },
      'Amazon Prime Video': { costo: 3000, valor: 4500, ciclo: 'Mensual' },
      'Paramount+': { costo: 3000, valor: 4000, ciclo: 'Mensual' },
      'Apple TV+': { costo: 3000, valor: 4500, ciclo: 'Mensual' },
      'Spotify': { costo: 4000, valor: 5500, ciclo: 'Anual' },
      'YouTube Premium': { costo: 3000, valor: 4500, ciclo: 'Mensual' },
      'Crunchyroll': { costo: 3000, valor: 4500, ciclo: 'Anual' },
      'ChatGpt (GO)': { costo: 5000, valor: 8000, ciclo: 'Mensual' },
    };
    const SERVICIOS_RESUMEN = ['Netflix', 'Disney+', 'HBO Max', 'Amazon Prime Video', 'Paramount+', 'Apple TV+', 'Spotify', 'YouTube Premium', 'Crunchyroll', 'Otro'];

    function showToast(msg) {
      const t = $('toast'); t.textContent = msg; t.classList.add('show');
      setTimeout(() => t.classList.remove('show'), 2200);
    }

    $('c_servicio').addEventListener('change', () => {
      const p = PRECIOS[$('c_servicio').value];
      if (p && !editCuentaId) {
        $('c_costo').value = p.costo; $('c_valor').value = p.valor; $('c_ciclo').value = p.ciclo;
      }
    });

    // ---------- STORAGE ----------
    async function loadData() {
      try {
        const r = await window.storage.get('cuentas');
        cuentas = r ? JSON.parse(r.value) : [];
      } catch (e) { cuentas = []; }
      try {
        const r2 = await window.storage.get('clientes');
        clientes = r2 ? JSON.parse(r2.value) : [];
      } catch (e) { clientes = []; }
      render();
    }
    async function saveCuentas() {
      try { await window.storage.set('cuentas', JSON.stringify(cuentas)); }
      catch (e) { showToast('⚠ Error guardando cuentas'); }
    }
    async function saveClientes() {
      try { await window.storage.set('clientes', JSON.stringify(clientes)); }
      catch (e) { showToast('⚠ Error guardando clientes'); }
    }

    // ---------- HELPERS DE CÁLCULO ----------
    function diffDias(fechaA, fechaB) {
      const a = new Date(fechaA); const b = new Date(fechaB);
      return Math.round((b - a) / 86400000);
    }
    function hoyISO() { return new Date().toISOString().slice(0, 10); }

    function calcularCuenta(c) {
      const hoy = hoyISO();
      c.ganancia = (Number(c.valor) || 0) - (Number(c.costo) || 0);
      c.duracion = (c.inicio && c.venc) ? diffDias(c.inicio, c.venc) : null;
      c.diasRestantes = c.venc ? diffDias(hoy, c.venc) : null;
      c.proximoPago = c.venc || null;
      if (c.diasRestantes === null) { c.estado = 'Sin fecha'; }
      else if (c.diasRestantes < 0) { c.estado = 'Vencida'; }
      else if (c.diasRestantes <= 3) { c.estado = 'Por vencer'; }
      else { c.estado = 'Activa'; }
      return c;
    }

    function cicloAMeses(ciclo) { return ciclo === 'Anual' ? 12 : 1; }

    function estadoBadge(estado) {
      const map = { 'Activa': 'activa', 'Por vencer': 'porvencer', 'Vencida': 'vencida', 'Sin fecha': 'offline' };
      return `<span class="badge ${map[estado] || 'offline'}">${estado}</span>`;
    }

    function fmtMoney(n) { return '$' + (Number(n) || 0).toLocaleString('es-CL'); }
    function fmtFecha(f) { if (!f) return '—'; const [y, m, d] = f.split('-'); return `${d}-${m}-${y}`; }

    // ---------- CUENTAS: CRUD ----------
    function limpiarFormCuenta() {
      ['c_perfil', 'c_usuario', 'c_pass', 'c_cliente', 'c_tel', 'c_inicio', 'c_venc', 'c_costo', 'c_valor', 'c_obs'].forEach(id => $(id).value = '');
      $('c_servicio').value = 'Netflix'; $('c_ciclo').value = 'Mensual'; $('c_online').value = 'Sí';
      editCuentaId = null;
      $('formCuentasTitle').textContent = 'Nueva cuenta';
      $('btnCancelarCuenta').classList.add('hidden');
    }

    $('btnGuardarCuenta').addEventListener('click', async () => {
      const cliente = $('c_cliente').value.trim();
      const usuario = $('c_usuario').value.trim();
      if (!cliente || !usuario) { showToast('⚠ Completa al menos Usuario/Correo y Cliente'); return; }

      let c = editCuentaId ? cuentas.find(x => x.id === editCuentaId) : { id: uid(), recordatorio: false };
      c.servicio = $('c_servicio').value;
      c.perfil = $('c_perfil').value.trim();
      c.usuario = usuario;
      c.pass = $('c_pass').value;
      c.cliente = cliente;
      c.tel = $('c_tel').value.trim();
      c.inicio = $('c_inicio').value;
      c.venc = $('c_venc').value;
      c.ciclo = $('c_ciclo').value;
      c.costo = $('c_costo').value;
      c.valor = $('c_valor').value;
      c.online = $('c_online').value;
      c.obs = $('c_obs').value;
      calcularCuenta(c);

      if (!editCuentaId) { cuentas.push(c); }
      await saveCuentas();
      ensureClienteExiste(cliente, c.tel);
      limpiarFormCuenta();
      render();
      showToast('✓ Cuenta guardada');
    });

    $('btnCancelarCuenta').addEventListener('click', limpiarFormCuenta);

    function editarCuenta(id) {

    function enviarWhatsApp(c) {

    function ensureClienteExiste(nombre, tel) {
      const existe = clientes.find(x => x.nombre.toLowerCase() === nombre.toLowerCase());

        clientes.push({ id: uid(), nombre, ig: '', tel: tel || '' });

    function statsCliente(nombre) {
      const propias = cuentas.filter(c => c.cliente.toLowerCase() === nombre.toLowerCase());
      const cantidad = propias.length;
      const gastoMensual = propias.reduce((sum, c) => {
        const meses = cicloAMeses(c.ciclo);
        return sum + ((Number(c.valor) || 0) / meses);
      }, 0);
      const vencimientos = propias.map(c => c.venc).filter(Boolean).sort();
      const proximoVenc = vencimientos[0] || null;
      let estadoGeneral = 'Sin cuentas';
      if (cantidad > 0) {
        if (propias.some(c => c.estado === 'Vencida')) estadoGeneral = 'Con cuentas vencidas';
        else if (propias.some(c => c.estado === 'Por vencer')) estadoGeneral = 'Por vencer pronto';
        else estadoGeneral = 'Al día';
      }
      return { cantidad, gastoMensual, proximoVenc, estadoGeneral };
    }

    function excelDate(iso) {
      if (!iso) return null;
      const d = new Date(iso + 'T00:00:00Z');
      const epoch = new Date(Date.UTC(1899, 11, 30));
      return Math.round((d - epoch) / 86400000);
    }

    $('btnExportar').addEventListener('click', () => {
      if (cuentas.length === 0 && clientes.length === 0) { showToast('⚠ No hay datos para exportar'); return; }

      const wb = XLSX.utils.book_new();
      const DATE_FMT = 'dd\\-mm\\-yyyy';
      const MONEY_FMT = '\\$#,##0';

      const headersCuentas = ['Servicio', 'Perfil', 'Usuario / Correo', 'Contraseña', 'Cliente', 'Teléfono Cliente',
        'Fecha de Inicio', 'Fecha de Vencimiento', 'Ciclo de Pago', 'Costo ($)', 'Valor ($)', 'Ganancia ($)',
        'Duración (días)', 'Días Restantes', 'Próxima Fecha de Pago', 'Estado', 'En línea', 'Recordatorio WhatsApp', 'Observaciones'];
      const wsC = XLSX.utils.aoa_to_sheet([headersCuentas]);
      for (let i = 0; i < 300; i++) {
        const r = i + 2;
        const c = cuentas[i];
        if (c) {
          XLSX.utils.sheet_add_aoa(wsC, [[
            c.servicio || '', c.perfil || '', c.usuario || '', c.pass || '', c.cliente || '', c.tel || '',
            null, null, c.ciclo || 'Mensual', Number(c.costo) || 0, Number(c.valor) || 0
          ]], { origin: `A${r}` });
          if (c.inicio) { wsC[`G${r}`] = { t: 'n', v: excelDate(c.inicio), z: DATE_FMT }; }
          if (c.venc) { wsC[`H${r}`] = { t: 'n', v: excelDate(c.venc), z: DATE_FMT }; }
          wsC[`Q${r}`] = { t: 'str', v: c.online || 'No' };
          wsC[`R${r}`] = { t: 'str', v: c.recordatorio ? 'Sí' : 'No' };
          wsC[`S${r}`] = { t: 'str', v: c.obs || '' };
        }
        wsC[`L${r}`] = { f: `IF(OR($E${r}="",$J${r}="",$K${r}=""),"",$K${r}-$J${r})`, z: MONEY_FMT };
        wsC[`M${r}`] = { f: `IF(OR($G${r}="",$H${r}=""),"",$H${r}-$G${r})` };
        wsC[`N${r}`] = { f: `IF($H${r}="","",$H${r}-TODAY())` };
        wsC[`O${r}`] = { f: `IF($H${r}="","",$H${r})`, z: DATE_FMT };
        wsC[`P${r}`] = { f: `IF($H${r}="","",IF($N${r}<0,"Vencida",IF($N${r}<=3,"Por vencer","Activa")))` };
      }

      XLSX.utils.book_append_sheet(wb, wsC, 'Cuentas');

      // ---- Hoja Clientes (60 filas de capacidad, cabecera en fila 3 igual que tu plantilla) ----
      const wsCl = XLSX.utils.aoa_to_sheet([
        ['Resumen por Cliente'],
        [],

        wsCl[`D${r}`] = { f: `IF($A${r}="","",COUNTIF(Cuentas!$E$2:$E$301,$A${r}))` };
        wsCl[`E${r}`] = { f: `IF($A${r}="","",SUMPRODUCT((Cuentas!$E$2:$E$301=$A${r})*N(Cuentas!$K$2:$K$301)/IF(Cuentas!$I$2:$I$301="Anual",12,1)))`, z: MONEY_FMT };
        wsCl[`F${r}`] = { f: `IF(OR($A${r}="",$D${r}=0),"",SUMPRODUCT(MIN(Cuentas!$H$2:$H$301+(Cuentas!$E$2:$E$301<>$A${r})*999999+(Cuentas!$H$2:$H$301="")*999999)))`, z: DATE_FMT };

        [],
        [],
        [],
        [],
        ['Cuentas por Servicio'],
        ['Servicio', 'Cantidad', 'Ingreso Total', 'Ganancia Total'],
      ]);
      wsR['A4'] = { f: 'COUNTIF(Cuentas!P2:P301,"Activa")' };
      wsR['C4'] = { f: 'COUNTIF(Cuentas!P2:P301,"Por vencer")' };
      wsR['E4'] = { f: 'COUNTIF(Cuentas!P2:P301,"Vencida")' };
      wsR['G4'] = { f: 'SUMIF(Cuentas!P2:P301,"Activa",Cuentas!K2:K301)', z: MONEY_FMT };
      wsR['I4'] = { f: 'SUMIF(Cuentas!P2:P301,"Activa",Cuentas!L2:L301)', z: MONEY_FMT };
      SERVICIOS_RESUMEN.forEach((serv, i) => {
